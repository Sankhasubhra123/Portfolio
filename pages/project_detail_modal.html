Project Overview

The E-commerce Platform represents a comprehensive full-stack solution designed to address the growing need for small to medium businesses to establish a robust online presence. This project emerged from identifying the gap between expensive enterprise solutions and basic template-based stores that lack customization and scalability.

Built with modern web technologies, the platform features a React-based frontend with server-side rendering capabilities, a Node.js/Express backend with RESTful API architecture, and PostgreSQL for reliable data persistence. The system handles everything from product catalog management to secure payment processing and real-time inventory tracking.

What sets this platform apart is its focus on performance optimization, achieving sub-2-second load times, and its comprehensive admin dashboard that empowers business owners to manage their operations without technical expertise. The project demonstrates proficiency in full-stack development, database design, payment integration, and modern deployment practices.

React Product Component

React 
const ProductCard = ({ product, onAddToCart }) => {
  const [isLoading, setIsLoading] = useState(false);
  const [imageLoaded, setImageLoaded] = useState(false);

  const handleAddToCart = async () => {
    setIsLoading(true);
    try {
      await onAddToCart(product.id);
      toast.success('Added to cart successfully!');
    } catch (error) {
      toast.error('Failed to add to cart');
    } finally {
      setIsLoading(false);
    }
  };

  return (
    <div className="card hover:shadow-hover transition-all duration-300">
      <div className="relative overflow-hidden rounded-lg mb-4">
        <img
          src={product.image}
          alt={product.name}
          className={`w-full h-48 object-cover transition-opacity duration-300 ${
            imageLoaded ? 'opacity-100' : 'opacity-0'
          }`}
          onLoad={() => setImageLoaded(true)}
          loading="lazy"
        />
        {!imageLoaded && (
          <div className="absolute inset-0 bg-secondary-100 animate-pulse" />
        )}
      </div>
      
      <h3 className="font-semibold mb-2">{product.name}</h3>
      <p className="text-text-secondary text-sm mb-4">{product.description}</p>
      
      <div className="flex items-center justify-between">
        <span className="text-xl font-bold text-primary">
          ${product.price}
        </span>
        <button
          onClick={handleAddToCart}
          disabled={isLoading}
          className="btn-primary disabled:opacity-50"
        >
          {isLoading ? 'Adding...' : 'Add to Cart'}
        </button>
      </div>
    </div>
  );
};

// POST /api/cart/add
router.post('/add', authenticateToken, async (req, res) => {
  const { productId, quantity = 1 } = req.body;
  const userId = req.user.id;

  try {
    // Check product availability
    const product = await Product.findById(productId);
    if (!product) {
      return res.status(404).json({ error: 'Product not found' });
    }

    if (product.stock < quantity) {
      return res.status(400).json({ 
        error: 'Insufficient stock',
        available: product.stock 
      });
    }

    // Add to cart or update quantity
    const cartItem = await CartItem.findOne({ 
      userId, 
      productId 
    });

    if (cartItem) {
      cartItem.quantity += quantity;
      await cartItem.save();
    } else {
      await CartItem.create({
        userId,
        productId,
        quantity,
        price: product.price
      });
    }

    // Update product stock
    product.stock -= quantity;
    await product.save();

    // Emit real-time update
    io.to(`user_${userId}`).emit('cartUpdated', {
      productId,
      quantity: cartItem ? cartItem.quantity : quantity
    });

    res.json({ 
      success: true, 
      message: 'Item added to cart' 
    });

  } catch (error) {
    console.error('Cart add error:', error);
    res.status(500).json({ error: 'Internal server error' });
  }
});
